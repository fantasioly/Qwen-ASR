```typescript
// GET /api/settings response
interface SettingsResponse {
  api_base_url: string;    // e.g. "http://10.50.193.74:30003/v1"
  api_key: string;         // masked: "***" if set, "" if empty
  port: number;            // e.g. 8000
  cors_origins: string;    // e.g. "http://localhost:5173"
  request_timeout: number; // e.g. 30
  model_name: string;      // e.g. "/bmcp_lvm_fs/cusa/models/Qwen3-ASR-1.7B"
}

// PUT /api/settings request body (any subset of above)
type UpdateSettings = Partial<Pick<SettingsResponse, 'api_base_url' | 'api_key' | 'port' | 'cors_origins' | 'request_timeout'>>;

// PUT /api/settings response
interface UpdateSettingsResponse {
  status: "ok";
  message: string;
  updated_keys: string[];  // e.g. ["API_BASE_URL", "REQUEST_TIMEOUT"]
}
```
